5.8. Deklaration der Einrichtstruktur für Anlagen-Globale Daten

In den Anlagen-Globalen-Daten werden alle Daten eingetragen, die Anlagenweit gültig sind. Das Feld "Reserve", wird zur Zeit nicht benötigt und ist mit NULL aufzufüllen. 

#define CODE_NR_LEN           6

typedef struct GlobAnlDaten
{
  char UserToUser[ USER_TO_USER_LEN + 1 ];      /* für beide Anlagen gleich     */
  char CodeNummer[ CODE_NR_LEN + 1 ];
  char Landeskennung[ CODE_NR_LEN + 1 ];
  char OrtsKennZif[ NAH_BER_NR_LEN + 1 ];
  char Hotline[ RUF_NR_LEN + 1 ];
  unsigned char Moh;
  unsigned char AmtAmtWeitergabe;
  unsigned char Reserve[ 0x1f ];
  unsigned char AktAva;
} GlobAnlDatenType;
