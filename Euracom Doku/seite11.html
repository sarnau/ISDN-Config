5.10. Deklaration der Struktur der Teilnehmerdaten

typedef struct TlnDataStruct
{
  unsigned char         PortNr; 			/* Zugehöriger Port     */
  unsigned long         Geb;
  unsigned char         IntRufNr;       		/* binär codierte Internrufnummer       */
  unsigned short        SchaltStatus;   		/* Status der Schalt-LMs        */
  unsigned short        WeckZeit;       		/* Weckzeit in Minuten  */
  unsigned short        Rechte; 			/* Berechtigungen des Teilnehmers       */
  unsigned short        Abteilung;      		/* Abrechnungstechnische Einheit        */
  unsigned short        Gruppe; 			/* Räumliche Einheit    */
  unsigned short        KW_Tab; 			/* Default: -1 für nicht belegt */
  unsigned short        CFB_Id[ RUF_NR_LEN + 1];        /* Default: -1 für nicht belegt */
  unsigned short        CFNR_Id[ RUF_NR_LEN + 1];       /* Default: -1 für nicht belegt */
  unsigned char         CFNR_Zeit;      		/* Default: 0 Sek für CFNR => CFU       */
  TYPE_TLN_TYP      Typ;       				 /* Teilnehmerklasse und Dienste */
} TlnDataType;
