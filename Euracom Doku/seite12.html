5.11. Deklaration der Fehlerspeicherauswertungsstruktur

#define SOS_ERROR_BUF_LEN    0x40

typedef struct SosFktErrorStruct
{
  unsigned short Info;  	/* Funktionskennung     */
  unsigned short LinNum;        /* Zeilennummer         */
  unsigned short ErrorC;        /* Fehlercode   */
  unsigned char Task;   /* Nummer des Prozesses */
} SosFktErrorType;

typedef struct ErrorStruct
{
  unsigned short ErrorCnt;
  struct SosFktErrorStruct Err[ SOS_ERROR_BUF_LEN ];
}ErrorType;

Struktur des Übertragungsrahmens

typedef struct RdErrorStruct
{
  unsigned short ErrorCnt;      /* Anzahl der Einträge in der Euracom   */
  struct SosFktErrorStruct Err[ 0x10 ]; /* maximal 10 Einträge  */
}RdErrorType;

6.1. Deklaration der Struktur der Kurzwahlziele

In der Kurzwahl Struktur werden alle Kurzwahlziele sowie die dazugehörigen Namenseinträge gespeichert. In die Euracom werden aber nur die Kurzwahlziele übertragen. 

#define ANZ_KW_NR  200

typedef struct KzWork
{
  struct KurzwZiele RNr[ ANZ_KW_NR ];
  struct KzNamen    Name[ ANZ_KW_NR ];
} KzWorkType;
